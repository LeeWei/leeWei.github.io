# 4. js 中 isNaN的使用

isNaN() 函数用于检查其参数是否是非数字值。

如果参数值为 NaN 或字符串、对象、undefined等非数字值则返回 true, 否则返回 false。

< body >
    <
    input type = "text"
id = "input" >
    <
    input type = "button"
value = "确定"
id = "btn" >
    <
    /body> <
    script >

    var box = document.getElementById('input');
var btn = document.getElementById('btn');
btn.onclick = function() {
        var val = box.value;
        val = Number(val);
        var res;
        // 1-
        res = isNaN(val) ? '非数字' : '数字';
        console.log(res);
        // 2--
        // isNaN(val) ? alert('非数字') : alert('数字');

        // if (!isNaN(val)) {
        //     console.log(true);
        // } else {
        //     console.log(false);
        // }
    } <
    /script>
