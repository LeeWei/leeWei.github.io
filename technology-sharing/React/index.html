# 子组件---->父组件传值

//  vue 中可以通过ref的形式直接修改父组件的参数信息不推荐，也可以使用this.$emit(arg1,arg2) arg1:方法名字，arg2:要传出的值 详情请参考vue组件传值--详细版
// react 父组件中通过给子组件绑定一个方法  子组件通过this.props.function 调用父组件传递的方法 父组件就可以接受子组件传递的值了,其实这块react 子传父和vue 的原理是一样的,只不过vue 帮我们封装好了这个方法

import React, { Component } from "react"
//  父组件
 class ParentList extends React.Component {
         constructor() {
             super()
             this.state = {
                name:"老张",
                age:"55"
             }
         };
         handleChange(e){
            this.setState({age:e.target.value}) 
         }
         handleParend(child){
            //  console.log(child)
             this.setState({age:child})
         }
         render() {
             return (
                 <div>
                    <h1>我是父组件</h1>
                    <p>姓名:{this.state.name}</p>
                    <p>年龄:{this.state.age}</p>
                    <input type="text" value={this.state.age} onChange={()=>this.handleChange(event)} />
                    <Childlist {...this.state}  handleParent={this.handleParend.bind(this)} />
                 </div>
             )
         }
    };
//  子组件组件
 class Childlist extends React.Component {
    constructor(props) {
        super()
        this.state = {
           name:"小张",
           age:props.age
        }
    };
    handleChange(e){
        this.setState({age:e.target.value}) 
        this.props.handleParent(e.target.value)
     }
    render() {
        return (
            <div>
               <h1>我是子组件</h1>
               <p>父组件姓名:{this.props.name}</p>//父组件到子组件通过props接受
               <p>父组件年龄:{this.props.age}</p>
               <p>子组件年龄:{this.state.age}</p>
               <input type="text" value={this.state.age} onChange={()=>this.handleChange(event)} />
            </div>
        )
    }
};
    ReactDOM.render(
        <div>
            <ParentList></ParentList>
        </div>,
        document.getElementById("my")
    )
